# RuntipiOS Boot Configuration for Raspberry Pi 5

# Enable 64-bit mode
arm_64bit=1

# GPU Memory
gpu_mem=128

# Enable UART
enable_uart=1

# Kernel
kernel=kernel8.img

# Device Tree overlays

# Disable splash screen

# Camera (disabled to save memory)

# Enable I2C and SPI (for future expansions)
dtparam=i2c_arm=on
dtparam=spi=on

# Audio
dtparam=audio=on
